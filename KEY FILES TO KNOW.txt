1. SAFE NUMBER (Prevent DB overflow - ALWAYS USE!):
```typescript
const safeNumber = (value: any, max = 99999999, decimals = 4): number => {
  if (!value) return 0;
  let num = parseFloat(String(value));
  if (isNaN(num) || !isFinite(num)) return 0;
  if (num > max) num = max;
  if (num < -max) num = -max;
  return Number(num.toFixed(decimals));
};
```

```typescript
// Format seconds to display
const formatTime = (seconds: number): string => {
  const h = Math.floor(seconds / 3600);
  const m = Math.floor((seconds % 3600) / 60);
  return `${h} h ${m} m`;
};

// Safe time conversion
const safeTimeSeconds = (value: any): number => {
  const num = parseInt(String(value)) || 0;
  const MAX = 365 * 24 * 60 * 60; // 1 year max
  return Math.min(Math.max(0, num), MAX);
};
```
